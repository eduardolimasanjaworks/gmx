  /**
   * Prepara um canvas binarizado (preto e branco) a partir de uma imagem
   * com um determinado fator de escala.
   */
  const buildBinarizedCanvas = (img: HTMLImageElement, scale = 2, threshold = 140) => {
    const canvas = document.createElement("canvas");
    const ctx = canvas.getContext("2d")!;
    canvas.width = img.width * scale;
    canvas.height = img.height * scale;
    ctx.filter = 'contrast(1.6) grayscale(1)';
    ctx.drawImage(img, 0, 0, canvas.width, canvas.height);
    const imageData = ctx.getImageData(0, 0, canvas.width, canvas.height);
    const d = imageData.data;
    for (let i = 0; i < d.length; i += 4) {
      const brightness = 0.34 * d[i] + 0.5 * d[i + 1] + 0.16 * d[i + 2];
      const color = brightness > threshold ? 255 : 0;
      d[i] = d[i + 1] = d[i + 2] = color;
    }
    ctx.putImageData(imageData, 0, 0);
    return canvas;
  };

  /**
   * Recorta uma zona percentual de um canvas já processado e faz zoom adicional.
   * @param src       Canvas processado (imagem inteira)
   * @param x0..y1   Coordenadas em fração da imagem original (0.0 a 1.0)
   * @param zoom     Fator extra de zoom para a zona
   */
  const cropZone = (src: HTMLCanvasElement, x0: number, y0: number, x1: number, y1: number, zoom = 3): string => {
    const sx = Math.floor(x0 * src.width);
    const sy = Math.floor(y0 * src.height);
    const sw = Math.floor((x1 - x0) * src.width);
    const sh = Math.floor((y1 - y0) * src.height);
    const canvas = document.createElement("canvas");
    canvas.width = sw * zoom;
    canvas.height = sh * zoom;
    const ctx = canvas.getContext("2d")!;
    ctx.imageSmoothingEnabled = false; // Nitidez pixel a pixel
    ctx.drawImage(src, sx, sy, sw, sh, 0, 0, canvas.width, canvas.height);
    return canvas.toDataURL("image/png");
  };

  const performOCR = async (file: File) => {
    setOcrLoading(true);
    try {
      // Carregar imagem original
      const img = new Image();
      const objectUrl = URL.createObjectURL(file);
      await new Promise<void>((resolve, reject) => {
        img.onload = () => resolve();
        img.onerror = reject;
        img.src = objectUrl;
      });
      URL.revokeObjectURL(objectUrl);

      // Canvas base binarizado em escala 2x (para o OCR geral)
      const baseCanvas = buildBinarizedCanvas(img, 2, 140);
      const fullImageUrl = baseCanvas.toDataURL("image/png");

      // ── Zonas específicas da CNH (frações x0, y0, x1, y1) ──────
      // Layout padrão CNH brasileira (horizontal, frente)
      const zones = {
        // Linha de datas (emissão, 1ª hab, validade) — faixa vertical 40-52%
        datas:     cropZone(baseCanvas, 0.10, 0.38, 0.90, 0.56, 3),
        // CPF e Nº Registro — faixa 52-65%
        cpf:       cropZone(baseCanvas, 0.10, 0.52, 0.75, 0.66, 3),
        // Categoria — canto inferior direito
        categoria: cropZone(baseCanvas, 0.75, 0.38, 1.00, 0.60, 4),
        // Filiação (nome da mãe) — faixa 65-80%
        filiacao:  cropZone(baseCanvas, 0.10, 0.64, 0.90, 0.82, 3),
        // RENACH / Formulário — rodapé
        rodape:    cropZone(baseCanvas, 0.10, 0.82, 0.90, 1.00, 3),
      };

      // Cria um único worker e reutiliza para todas as zonas
      const worker = await createWorker('por');

      const [fullText, datasText, cpfText, catText, filiacaoText, rodapeText] = await Promise.all([
        worker.recognize(fullImageUrl).then(r => r.data.text),
        worker.recognize(zones.datas).then(r => r.data.text),
        worker.recognize(zones.cpf).then(r => r.data.text),
        worker.recognize(zones.categoria).then(r => r.data.text),
        worker.recognize(zones.filiacao).then(r => r.data.text),
        worker.recognize(zones.rodape).then(r => r.data.text),
      ]);

      await worker.terminate();

      // Retorna o texto completo como base, mais as zonas para o parser especializado
      return {
        full: fullText,
        datas: datasText,
        cpf: cpfText,
        categoria: catText,
        filiacao: filiacaoText,
        rodape: rodapeText,
      };
    } catch (error) {
      console.error("OCR Error:", error);
      throw new Error("Falha ao processar OCR (tesseract)");
    } finally {
      setOcrLoading(false);
    }
  };

  const parseCnhOcrData = (ocrResult: string | { full: string; datas: string; cpf: string; categoria: string; filiacao: string; rodape: string }, currentForm: any) => {
    const updates: any = {};

    // Suporta tanto string simples (legado) quanto objeto de zonas
    const isZoned = typeof ocrResult === 'object';
    const text     = isZoned ? ocrResult.full : ocrResult;
    const textDatas    = isZoned ? ocrResult.datas    : ocrResult;
    const textCpf      = isZoned ? ocrResult.cpf      : ocrResult;
    const textCategoria = isZoned ? ocrResult.categoria : ocrResult;
    const textFiliacao = isZoned ? ocrResult.filiacao  : ocrResult;
    const textRodape   = isZoned ? ocrResult.rodape    : ocrResult;
    
    // Corrige erros bizarros de OCR em números (ex: o723180194o -> 07231801940)
    const fixOcr = (t: string) => t.replace(/[oO](?=\d)|(?<=\d)[oO]/g, '0');
    const norm  = fixOcr(text).replace(/\s+/g, ' ').toUpperCase();
    const lines = text.split('\n').map((l: string) => l.trim()).filter(Boolean);

    // ── DATAS: Usa a zona de datas para maior precisão ──────────────
    const textNumericoDatas = fixOcr(textDatas);
    const dateRegex = /\b(\d{1,2})\s*[\/.-]\s*(\d{1,2})\s*[\/.-]\s*(\d{4})\b/g;
    const allDates: Array<{str: string, ts: number}> = [];
    let match;
    while ((match = dateRegex.exec(textNumericoDatas)) !== null) {
      const d = parseInt(match[1], 10);
      const m = parseInt(match[2], 10);
      const y = parseInt(match[3], 10);
      const ts = new Date(y, m - 1, d).getTime();
      if (ts > new Date(1940, 0, 1).getTime() && ts < new Date(2100, 0, 1).getTime()) {
        const formatted = `${match[1].padStart(2, '0')}/${match[2].padStart(2, '0')}/${match[3]}`;
        allDates.push({ str: formatted, ts });
      }
    }

    const uniqueDates = Array.from(new Map(allDates.map(item => [item.ts, item])).values());
    uniqueDates.sort((a, b) => a.ts - b.ts);

    const now = Date.now();
    const past = uniqueDates.filter(d => d.ts <= now);
    const future = uniqueDates.filter(d => d.ts > now);

    if (!currentForm.validade && future.length > 0) updates.validade = future[future.length - 1].str;

    if (past.length >= 3) {
      if (!currentForm.data_nasc) updates.data_nasc = past[0].str;
      if (!currentForm.primeira_habilitacao) updates.primeira_habilitacao = past[1].str;
      if (!currentForm.emissao_cnh) updates.emissao_cnh = past[past.length - 1].str;
    } else if (past.length === 2) {
      if (!currentForm.data_nasc) updates.data_nasc = past[0].str;
      if (!currentForm.emissao_cnh) updates.emissao_cnh = past[1].str;
    } else if (past.length === 1) {
      if (!currentForm.emissao_cnh) updates.emissao_cnh = past[0].str;
    }

    // ── CPF: usa zona dedicada de CPF ──────────────────────────────
    if (!currentForm.cpf) {
      const normCpf = fixOcr(textCpf).replace(/\s+/g, ' ').toUpperCase();
      const cpfMatch = normCpf.match(/\d{3}[\.\s]?\d{3}[\.\s]?\d{3}[-\s]?\d{2}/);
      if (cpfMatch) updates.cpf = cpfMatch[0].replace(/[^\d]/g, '');
    }

    // ── CATEGORIA: usa a zona do canto direito isolada ─────────────
    if (!currentForm.categoria) {
      // O texto da zona de categoria tem muito menos lixo, basta pegar a 1ª letra sozinha válida
      const normCat = textCategoria.replace(/\s+/g, ' ').toUpperCase().trim();
      // Remove palavras conhecidas que não são categorias
      const cleanedCat = normCat
        .replace(/CAT\.?\s*HAB|CATEGORIA|HABILITA[CÇ][AÃ]O|N[º°]?\s*REG|RENACH|VALIDADE|EM[IÍ]SS[AÃ]O/gi, '')
        .replace(/\d+/g, '') // remove números
        .replace(/[^A-Z\s]/g, '') // só letras
        .trim();
      // Pega apenas tokens de 1-2 letras que sejam categorias válidas
      const VALID_CATS = ['AB','AC','AD','AE','A','B','C','D','E'];
      const IGNORE_TOKENS = new Set(['DE','DA','DO','AS','OS','SE','EM','AO','OU','NO','NA','HA']);
      const tokens = cleanedCat.split(/\s+/);
      const catFound = tokens.find(t => VALID_CATS.includes(t) && !IGNORE_TOKENS.has(t));
      if (catFound) updates.categoria = catFound;
    }

    // ── Nº REGISTRO CNH (11 dígitos) ───────────────────────────────
    if (!currentForm.n_registro_cnh) {
      const cpfDigits = updates.cpf || currentForm.cpf || '';
      const regMatches = [...norm.matchAll(/\b(\d{11})\b/g)].map(x => x[1]);
      const registro = regMatches.find(n => n !== cpfDigits);
      if (registro) updates.n_registro_cnh = registro;
    }

    // ── Nº FORMULÁRIO / ESPELHO (9 a 12 dígitos) ───────────────────
    if (!currentForm.n_formulario_cnh) {
      const formMatches = [...norm.matchAll(/\b(\d{9,12})\b/g)].map(x => x[1]);
      const cpfDigits = updates.cpf || currentForm.cpf || '';
      const regDigits = updates.n_registro_cnh || currentForm.n_registro_cnh || '';
      const formulario = formMatches.find(n => n !== cpfDigits && n !== regDigits);
      if (formulario) updates.n_formulario_cnh = formulario;
    }

    // -- RENACH (UF + 9 digitos): prioriza zona do rodape --
    if (!currentForm.n_cnh_renach) {
      const renachSources = [textRodape, text];
      for (const src of renachSources) {
        const renachRaw = src.toUpperCase();
        const renachMatch = renachRaw.match(/\b([A-Z]{2}[0-9A-Z]{9})/);
        if (renachMatch) {
          const r = renachMatch[1];
          const uf = r.slice(0, 2);
          const num = r.slice(2).replace(/[O]/g, '0').replace(/[I]/g, '1').replace(/[S]/g, '5').replace(/[B]/g, '8').replace(/[Z]/g, '2');
          updates.n_cnh_renach = uf + num;
          break;
        }
      }
    }
    // -- NOME DA MAE: prioriza zona de filiacao dedicada --
    if (!currentForm.nome_mae) {
      // Estrategia 1: usa a zona recortada de filiacao (mais limpa)
      const filLines = textFiliacao.split('\n').map((l: string) => l.trim()).filter(Boolean);
      const filParentLines = filLines
        .map(l => l.replace(/[^a-zA-Z\u00C0-\u00FA\s]/g, '').trim())
        .filter(l => l.length > 5)
        .map(l => l.replace(/^(?:[a-zA-Z]{1,2}\s+)+/, '').trim());
      if (filParentLines.length >= 2) {
        updates.nome_mae = filParentLines[1];
      } else if (filParentLines.length === 1) {
        updates.nome_mae = filParentLines[0];
      } else {
        // Estrategia 2: fallback no texto completo
        const filiacaoIdx = lines.findIndex((l: string) => /FILIA[CÇ][AÃ]O|FIA[CÇ][AÃ]O|M[AÃ]E/i.test(l));
        if (filiacaoIdx !== -1) {
          const parentLines = lines.slice(filiacaoIdx + 1, filiacaoIdx + 5)
            .map(l => l.replace(/[^a-zA-Z\u00C0-\u00FA\s]/g, '').trim())
            .filter(l => l.length > 5)
            .map(l => l.replace(/^(?:[a-zA-Z]{1,2}\s+)+/, '').trim());
          if (parentLines.length >= 2) updates.nome_mae = parentLines[1];
          else if (parentLines.length === 1) updates.nome_mae = parentLines[0];
        }
      }
    }

    return {
      ...currentForm,
      ...Object.fromEntries(Object.entries(updates).filter(([, v]) => v !== null && v !== '')),
      observacao: currentForm.observacao
        ? currentForm.observacao + '\n\n[OCR]: ' + text
        : '[OCR]: ' + text,
    };
  };
